    13.                     // Owl Carousel Slide
$(document).ready(function() {
  $(".owl-carousel").owlCarousel({
    items: 4,
    margin: 10,
    nav: false, // This will an arrow tag in the slider
    loop: true, // This will help with the repeating of the slider
    autoplay: true, 
    autoplayTimeout: 2000,
    autoplayHoverPause: true,
    responsive: { // This is used for the reponsive
      0: {
        items: 1,
        nav: false
      },
      600: {
        items: 2,  // This determine the number of items that used slide if the screen size is reduce
        nav: false
      },
      1000: {
        items: 3,   // This tell the number of slides which should slide if its in full screen mode
        nav: false,
        loop: true
      }
    }
  });
});
